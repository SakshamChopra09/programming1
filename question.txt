// queue using LinkedList Enqueue-O(1)  from front and Dequeue from NULL pointer

#include <iostream>
struct Node
{
    int data;
    Node *next;
};
Node *front = NULL;

void Enqueue(int x)
{
    Node *temp = new Node();
    temp->data = x;
    temp->next = NULL;
    if (front == NULL)
    {
        front = temp;
        return;
    }
    temp->next = front;
    front = temp;
}

void Dequeue()
{
    Node *temp = front;
    Node *rear;
    while (temp != NULL)
    {
        rear = temp;
        temp = temp->next;
    }
    std::cout << rear->data << "\n";
    rear->next = NULL;
}

void Print()
{
    Node *temp = front;
    while (temp != NULL)
    {
        std::cout << temp->data << "  ";
        temp = temp->next;
    }
    std::cout << std::endl;
}

int main()
{
    Enqueue(2);
    Print();
    Enqueue(4);
    Print();
    Enqueue(6);
    Print();
    Dequeue();
    Print();
    Enqueue(8);
    Print();
    return 0;
}
